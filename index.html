function getSpeech(m) {
  const arr = moodMessages[m] || moodMessages.normal;
  return arr[Math.floor(Math.random() * arr.length)];
}

function getMoodIcon(m) {
  return { normal:'🐶', happy:'🥰', sulky:'😤', sleeping:'😴', angry:'😡', love:'💕' }[m] || '🐶';
}

function render() {
  // determine mood from stats
  if(state.mood !== 'sleeping') {
    if(state.love >= 85)           state.mood = 'love';
    else if(state.happy >= 80)     state.mood = 'happy';
    else if(state.happy <= 30)     state.mood = 'angry';
    else if(state.hunger <= 20 || state.happy <= 50) state.mood = 'sulky';
    else                           state.mood = 'normal';
  }

  drawDog(state.mood);
  document.getElementById('moodIcon').textContent = getMoodIcon(state.mood);
  document.getElementById('bubble').textContent   = getSpeech(state.mood);
  updateBars();
  checkAnniversary();
  updateDate();
}

// ── DATE ──
function updateDate() {
  const now = new Date();
  const start = new Date(now.getFullYear(), 3, 9); // April 9
  if(now < start) start.setFullYear(start.getFullYear()-1);
  const days = Math.floor((now - start) / 86400000);
  document.getElementById('dateBadge').textContent =
    `♡ 在一起第 ${days} 天`;
}

function checkAnniversary() {
  const now = new Date();
  const banner = document.getElementById('annivBanner');
  if(now.getMonth() === 3 && now.getDate() === 9) {
    banner.style.display = 'block';
  }
}

function showNotif(msg) {
  const n = document.getElementById('notif');
  n.textContent = msg;
  n.classList.add('show');
  setTimeout(() => n.classList.remove('show'), 2000);
}

function animatePet(cls) {
  const c = document.getElementById('petCanvas');
  c.classList.remove('pet-bounce','pet-shake');
  void c.offsetWidth;
  c.classList.add(cls);
  setTimeout(() => c.classList.remove(cls), 700);
}

async function sendChat() {
  const input = document.getElementById('chatInput');
  const btn   = document.getElementById('chatSend');
  const resp  = document.getElementById('chatResponse');
  const msg   = input.value.trim();
  if(!msg) return;

  btn.disabled = true;
  btn.textContent = '…';
  resp.style.display = 'block';
  resp.textContent = 'present 正在想…';

  const systemPrompt = `你扮演一个叫 present 的电子虚拟宠物小狗。
你的主人叫"纠缠"，你非常喜欢他/她但嘴硬不承认。
你的性格：超级可爱、玻璃心、经常表达爱、容易生闷气、很喜欢骂他、偶尔耍小脾气、爱撒娇但会假装不在乎。
当前状态：饥饿度${state.hunger}%，心情${state.happy}%，精力${state.energy}%，爱意${state.love}%。
回复要求：用简短可爱的中文回复（30字以内），要符合你的傲娇玻璃心性格，偶尔带上小情绪。
不要用markdown，不要解释自己在扮演角色，直接用角色的口吻说话。`;

  try {
    const response = await fetch('https://api.anthropic.com/v1/messages', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        model: 'claude-sonnet-4-20250514',
        max_tokens: 1000,
        system: systemPrompt,
        messages: [{ role: 'user', content: `纠缠对你说："${msg}"` }]
      })
    });
    const data = await response.json();
    const text = data.content?.map(b => b.text || '').join('') || '……';
    resp.textContent = text;
    document.getElementById('bubble').textContent = text.slice(0, 40);
    // boost love a little for talking
    state.love = clamp(state.love + 5);
    updateBars();
  } catch(e) {
    resp.textContent = '（网络不好，present 没收到你的消息，很委屈）';
  }

  btn.disabled = false;
  btn.textContent = '发送';
  input.value = '';
}

document.getElementById('chatInput').addEventListener('keydown', e => {
  if(e.key === 'Enter') sendChat();
});
